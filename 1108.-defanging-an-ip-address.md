# 1108. Defanging an IP Address

```cpp
class Solution {
public:
  
  vector<string> stringTokenizer(string s,char delimeter)
{
    stringstream ss(s); //convert my_string into string stream

    vector<string> tokens;
    string temp_string;

    while (getline(ss, temp_string, delimeter))
    { //use comma as delim for cutting string
        tokens.push_back(temp_string);
    }

    return tokens;
}
    string defangIPaddr(string address) {
      
      vector<string> tokens = stringTokenizer(address,'.');
      string res ;
      
      for(auto token : tokens) cout<<token<<" ";
      cout<<endl;
      
      res = res + tokens[0]+"[.]"+tokens[1]+"[.]"+tokens[2]+"[.]"+tokens[3];
      
      return res;
      
      
    }
};
```
