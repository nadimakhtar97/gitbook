# 1313. Decompress Run-Length Encoded List

```cpp
class Solution {
public:
    vector<int> decompressRLElist(vector<int>& nums) {
      
      vector<int> freq(101,-1);
      vector<int> res;
      
      for(int i=0;i<nums.size();i+=2)
      {
        int val = nums[i+1] , freq = nums[i];
        for(int j=0;j<freq;j++)
          res.push_back(val);
      }
      
      return res;
      
        
    }
};
```
