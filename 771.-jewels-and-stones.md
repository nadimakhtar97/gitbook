# 771. Jewels and Stones

```cpp
class Solution {
public:
    int numJewelsInStones(string jewels, string stones) {
      
      set<char> sjewel(jewels.begin(),jewels.end());
      int count = 0;
      
      for(auto it : sjewel)
        cout<<it<<" ";
        
        cout<<endl;
      
      for(auto c : stones)
      {
        if(sjewel.count(c)) count++;
      }
      
      return count;
        
    }
};
```
