# 1678. Goal Parser Interpretation

```cpp
class Solution {
public:
    string interpret(string command) {
      
      vector<char> cmd(command.begin(),command.end());
      string out = "";
      int i = 0;
      
      while(i<cmd.size())
      {
        if(cmd[i] == 'G') 
        {
          out += "G";  i++;
        }
        else if(cmd[i]=='('&&cmd[i+1]==')')
        {
          out+= "o"; i+=2;
        }
        else if(cmd[i] == '(' && cmd[i+1] == 'a') 
        {
          out+="al"; i+=4;
        }
          
      }
      
      cout<<out<<endl;
      
      return out;
        
    }
};
```
