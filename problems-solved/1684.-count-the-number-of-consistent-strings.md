# 1684. Count the Number of Consistent Strings

```cpp
class Solution {
public:
    int countConsistentStrings(string allowed, vector<string>& words) {
      
      
      bool allowedChar[123] = {};
      int res = words.size();
      
      
      for(auto c : allowed) allowedChar[c] = true;
      for(auto word : words)
      {
        for(auto c : word)
        {
          if(!allowedChar[c])
          {
            res--;
            break;
          }            
        }
      }
      
      return res;
      
    }
  
      
};
```
