# 804. Unique Morse Code Words

```cpp
class Solution {
public:
    int uniqueMorseRepresentations(vector<string>& words) {
      
      
      vector<string> morse = {".-", "-...", "-.-.", "-..", ".", "..-.", "--.", "....", "..", ".---", "-.-", ".-..", "--", "-.", "---", ".--.", "--.-", ".-.", "...", "-", "..-", "...-", ".--", "-..-", "-.--", "--.."};
      
      set<string> s;
      
      for(auto word :words)
      {
        string morseCode;
        for(auto letter : word) morseCode += morse[letter-97];
        s.insert(morseCode);
      }
      

      
      return s.size();
      
        
    }
};
```
