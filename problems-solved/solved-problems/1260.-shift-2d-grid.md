# 1260. Shift 2D Grid

```cpp
class Solution {
public:
    vector<vector<int>> shiftGrid(vector<vector<int>>& grid, int k) {
      
      int m = grid.size(),n = grid[0].size();
      
      vector<vector<int>> g(m,vector<int>(n,0));
      
      int shift = k%(n*m);
      
      for(int i=0;i<m;i++)
      {
        for(int j=0;j<n;j++)
        {
          int row = (i+(j+k)/n)%m;
          int column = (j+k)%n;
          g[row][column] = grid[i][j];
        }
      }
      
      return g;
        
    }
};
```
