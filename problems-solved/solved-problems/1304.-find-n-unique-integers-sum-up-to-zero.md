# 1304. Find N Unique Integers Sum up to Zero

```cpp
class Solution {
public:
    vector<int> sumZero(int n) {
      
      vector<int> out;
      
      if((n&1)!=0) out.push_back(0);
        for(int i=1;2*i<=n;i++)
        {
          out.push_back(i);
          out.push_back(-i);
        }
          
      for(auto o : out) cout<<o<<" ";
      
      return out;
        
      }
        
    
};
```
