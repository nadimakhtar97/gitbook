# 1332. Remove Palindromic Subsequences

```cpp
class Solution {
public:
  
    bool palindrome(string& s)
    {
      int i=0,j=s.size()-1;
      bool pldm = true;
      
      while(i<=j)
      {
        if(s[i++]!=s[j--])
          pldm = false;
      }
      
      return pldm;
    }
    int removePalindromeSub(string s) {
      
      
      if(s == "") return 0;
      
      if(palindrome(s))
        return 1;
      
      
      
        return 2;
        
    }
};
```
