# 1370. Increasing Decreasing String

```cpp
class Solution {
public:
    string sortString(string s) {
      
      
      int n = s.length();
      vector<int> c(123,0);
      string str = "";
      for(int i=0;i<n;i++)
      {
        c[s[i]]++;
      }
      
     
      while(str.length()<n)
      {
        for(int i=97;i<123;i++)
        {
          if(c[i])
          {
            str += (char) i;
            c[i]--;
          }
        }
      
      for(int i=122;i>96;i--)
        {
          if(c[i])
          {
            str += (char) i;
            c[i]--;
          }
        }
      }
      
      
      return str;
      
        
    }
};
```
