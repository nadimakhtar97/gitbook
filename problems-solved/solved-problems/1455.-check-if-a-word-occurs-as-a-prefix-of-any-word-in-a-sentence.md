# 1455. Check If a Word Occurs As a Prefix of Any Word in a Sentence

```cpp
class Solution {
public:
    int isPrefixOfWord(string sentence, string searchWord) {
      
      stringstream ss(sentence);
      string word;
      int len = searchWord.size();
      int pos = 1;
      bool found = false;
        
      while(ss >> word)
      {
        
        if(word.substr(0,len) == searchWord)
        {
          found = true;
          break;
        }
        pos++;
      }
      
      return found?pos:-1;
        
    }
};
```
