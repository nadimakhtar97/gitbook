# 1763. Longest Nice Substring

```cpp
class Solution {
public:
  
    bool isNice(string s)
    {
      for(auto c : s)
      {
        if( isupper(c)) 
        {
            if(s.find(tolower(c))==string::npos)
                return false;
        }else
        {
          if(s.find(toupper(c))==string::npos)
            return false;
        }
      }
      
      return true;
    }
    string longestNiceSubstring(string s) {
      
      int n =s.size();
      string niceString = "";
      
      
      for(int i = 2;i<=n;i++)
      {
        int low = 0;
        int high = i;
        
        while(high <= n)
        {
          string subString = s.substr(low,(high-low));
          // cout<<"substring = "<<subString<<endl;
          // cout<<"is nice = "<<isNice(subString)<<endl;
          if(isNice(subString))
          {
            if(subString.length() > niceString.length())
            niceString = subString;
          }
          high++;
          low++;
        }
      }
      
      return niceString;
      
      
        
    }
};
```
