# 1768. Merge Strings Alternately

```cpp
class Solution {
public:
    string mergeAlternately(string w1, string w2) {
      
      string mergedString = "";
      int n = w1.size(), m = w2.size(),i=0,j=0;
      
      while ( i<n || j<m)
      {
        if(i<n)
        mergedString += w1[i++];
        if(j<m)
        mergedString += w2[j++];
      }
      
      return mergedString;
      
      
        
    }
};
```
