# 575. Distribute Candies

```cpp
class Solution {
public:
    int distributeCandies(vector<int>& v) {
      
      set<int> candySet(v.begin(),v.end());
      int n = v.size()/2;
      int res = 0;
      
      if(candySet.size() >= n )
        res = n;
      else
        res = candySet.size();
      
      return res;
      
        
    }
};
```
