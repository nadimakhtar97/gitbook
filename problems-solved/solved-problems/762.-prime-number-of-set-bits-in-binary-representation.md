# 762. Prime Number of Set Bits in Binary Representation

```cpp
class Solution {
public:
  
    bool isPrime(int n)
    {
       if(n==0 || n==1) return false;
      
        for(long long i=2;i*i<=n;i++)
          if(n%i==0) return false;
        
      return true;
      
    }
  
    int countPrimeSetBits(int L, int R) {
      
      int count = 0;
      
      for(int i = L ;i<=R;i++)
      {
        bitset<32> b(i);
        if(isPrime(b.count())) count++;
        
      }
      
      return count;
        
    }
};
```
