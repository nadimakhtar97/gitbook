# 884. Uncommon Words from Two Sentences

```cpp
class Solution {
public:
    vector<string> uncommonFromSentences(string A, string B) {
      
      string AB(A+" "+B);
      stringstream ab(AB);
      vector<string> res;
      map<string,int> freq;
      string temp;
    
      
      while(ab >> temp)
      {
          freq[temp]++;
      }
      
      for(auto p : freq)
      {
        if(p.second == 1)
        res.push_back(p.first);
      }
      
      return res;
        
    }
};
```
