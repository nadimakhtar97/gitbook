# 985. Sum of Even Numbers After Queries

```cpp
class Solution {
public:
    vector<int> sumEvenAfterQueries(vector<int>& A, vector<vector<int>>& queries) {
      
      
      vector<int> res;
        int eSum = 0;
      
      
      
      for(int i = 0;i<A.size();i++)
        {
          if((A[i]&1) == 0) 
            eSum+=A[i];
        }
      
      for(auto& query : queries)
      {
        if((A[query[1]]&1)==0) 
        {
          eSum -= A[query[1]];
          A[query[1]] += query[0];
          if((A[query[1]]&1)==0)
          {
            eSum += A[query[1]];
            res.push_back(eSum);
          }
          else
          res.push_back(eSum);
          
        }
         else if((A[query[1]]&1)==1)
        {
          A[query[1]] += query[0];
          if((A[query[1]]&1)==0)
          {
            eSum += A[query[1]];
            res.push_back(eSum);
          }
          else
          res.push_back(eSum);
        }
        
        
      }
      
      return res;
        
    }
};
```
